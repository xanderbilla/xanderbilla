2] Write a program for Binary Search.

    
    def binary_search(l, n):            
        low = 0                
        high  = len(l) - 1
        mid = 0
        while low <= high :
        mid = (high + low) // 2                
        if l[mid] < n:                
            low = mid + 1                
        elif l[mid] > n:                
            high = mid - 1                
        else:                
            return mid                
    return -1
                                        
    l = [10,20,21,45,56,45]                
    l.sort()                
    n = 21                
    result = binary_search(l, n)            
                        
    if result != -1:                
        print("Element {} present at index {}".format(n,result))                
    else:                
        print("Element not found")

3] Write a program for Bubble Sort.

                                   
    def bubble_sort(l):                
        for i in range(len(l) - 1):                
            for j in range(len(l) - 1):                
                if (l[j] > l [j + 1]):                
                    temp = l[j]                
                    l[j] = l[j + 1]                
                    l[j + 1] = temp                
        return l         
        
    l = [25,23,32,85,77,45,12]                
    print("Unsorted list: ", l)                
    print("Sorted list: ", bubble_sort(l))

    a = int(input("Enter I number: "))

    while(a != b):                
        if (a > b):                
            a -= b

            b -= a

    def push():                
        global top                
        if (top == size - 1):                
            print("Stack Overflow")  
    
        top += 1                
        item = int(input("Enter a number to be inserted: "))

        global top                
        if (top == -1):                
            print("Stack Underflow")              
        
    
        print("The element {} is removed".format(x[top]))
