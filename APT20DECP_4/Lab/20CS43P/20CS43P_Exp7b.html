b.	write a JAVA program to implement Inheritance and demonstrate the 
    use of method overriding
**************************************************************************/

class parent {
    int a,b;

    void set() {
        a=10;
        b=20;
    }

    void compute()  {

    int c=a+b;
    System.out.println("sum of a and b="+c);
    }
}

class child extends parent {
    int x,y;

    void set1() {
        x=100;
        y=200;
    }
    
    void compute()  {
        int z= x*y;
        System.out.println("in child class");
        System.out.println("z="+z);
    }
}

class Exp7b {
    public static void main(String args[])
    {
        child C1=new child();
        C1.set();
        C1.set1();
        C1.compute();
    }
}
