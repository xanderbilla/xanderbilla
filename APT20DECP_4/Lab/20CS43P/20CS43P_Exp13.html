 processing and check compliance with SRP.

class Add_Calculator { // this class is only responsible for adding numbers
    public static int add(int x, int y) {
      return x + y;
    }
}
 
class Sub_Calculator { // this class is only responsible for subtracting numbers
    public static int sub(int x, int y) {
      return x - y;
    }
}

class Mul_Calculator { // this class is only responsible for multiplying numbers
    public static int mul(int x, int y) {
      return x * y;
    }
}

class Div_Calculator { // this class is only responsible for dividing numbers
    public static int div(int x, int y) {
      return x / y;
    }
}

class Exp9{
    public static void main(String[] args){
        System.out.println("The sum of 100 and 20 is: " + Add_Calculator.add(100, 20));
        System.out.println("The difference of 100 and 20 is: " + Sub_Calculator.sub(100, 20));
        System.out.println("The multiplication of 100 and 20 is: " + Mul_Calculator.mul(100, 20));
        System.out.println("The division of 100 and 20 is: " + Div_Calculator.div(100, 20));
    }
}
----------------------------------Output: ----------------------------- 
        
The sum of 100 and 20 is: 120
The difference of 100 and 20 is: 80
The multiplication of 100 and 20 is: 2000
The division of 100 and 20 is: 5
