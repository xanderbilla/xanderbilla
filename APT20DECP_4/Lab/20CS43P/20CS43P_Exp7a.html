    Concpet)
**************************************************************************/

class parents {
    int a;

    void get_m (int m)  {
        a = m;  
    }

    void display_m() {
        System.out.println("value of a is " + a);
    }
}

class child extends parents {
    int b;

    void get_n(int n)
    {
        b = n;
    }

    void display_n() {
        System.out.println("Value of b is " + b);
    }
}

class Exp7a {
    public static void main(String args[]) {
    child C1 = new child();
    C1.get_m(10);
    C1.get_n(20);
    C1.display_m();
    C1.display_n();
    }
}
