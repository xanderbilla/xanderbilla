Exp7: Program to implement Selection Sort, compute space time complexity and plot graph using Asymptotic notations.

            
#**********************************************************************************#
# Program to implememnt Selection Sort                                                #
#**********************************************************************************#

n = int(input("Enter number of element: "))
x = []
for i in range(n):
    x.append(int(input("Enter element: ")))
print(x)
for i in range(n-1):
    small = i
    for j in range(i + 1, n):
        if(x[j] < x[small]):
            small = j
    temp = x[i]
    x[i] = x[small]
    x[small] = temp
print(x)
