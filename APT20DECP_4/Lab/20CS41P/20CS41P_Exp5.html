def bsearch(low, high):
    global key
    if (low > high):
        return -1
    mid = (low + high) // 2
    if (key == x[mid]):
        return mid
    elif (key > x[mid]):
        return bsearch(mid + 1, high)
    else:
        return bsearch(low, high - 1)
