Exp6: Program to implement Insertion Sort, compute space time complexity and plot graph using Asymptotic notations.

            
#**********************************************************************************#
# Program to implememnt Insertion Sort                                                #
#**********************************************************************************#

n = int(input("Enter number of element: "))
x = []
for i in range(n):
    x.append(int(input("Enter element: ")))
print(x)
for i in range(1,n):
    temp = x[i]
    for j in range(i-1, -2, -1):
        if (temp < x[j]):
            x[j+1] = x[j]
        else:
            break
    x[j+1] = temp
print(x)
