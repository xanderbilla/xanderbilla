def push():
    global top 
    if (top == size - 1):
        print("Stack Overflow")
    
    top += 1
    item = int(input("Enter a number to be inserted: "))
    x.insert(top, item)

def pop():
    global top
    if (top == -1):
        print("Stack Underflow")

    print("The element {} is removed".format(x[top]))
    x.pop())
    top -= 1

def display():
    global top
    if (top == -1):
        print("Stack is empty")
    print(x)


x = []
top = -1
size = 5

while True:
    print("""
    
    Enter your choice:

    1) Insert element in stack
    2) Remove element from stack
    3) Display the stack
    other) exit

    """)  
    ch = int(input("Enter your choice: "))
    if ch == 1:
        push()
    elif ch == 2:
        pop()
    elif ch == 3:
        display()
    else:
        print("Invalid option")
        exit(0)
