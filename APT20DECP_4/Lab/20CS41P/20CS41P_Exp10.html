def q_insert():
    global rear
    if(rear == size-1):
        print("queue is full \n")
        return

    item = int(input("Enter item to be  inserted "))
    rear += 1
    x.insert(rear, item)

def q_delete():
    global front
    if(front > rear):
        print("Queue is empty")
        return
    print("Deleted element is {}".format(x[front]))
    front += 1



def display():
    if(front > rear):
        print("Queue is empty")
        return
    for i in range(front, rear + 1):
        print(x[i],"\n")


x = []
size = 5
rear = -1
front = 0

while True:
    print("""
    
    Enter your choice:

    1) Insert element in stack
    2) Remove element from stack
    3) Display the stack
    other) exit

    """)  
    ch = int(input("Enter your choice: "))
    if ch == 1:
        q_insert()
    elif ch == 2:
        q_delete()
    elif ch == 3:
        display()
    else:
        print("Invalid option")
        exit(0)
