x = [2,6,3,4,8,9,1,7,8,5,8]
print("The list in the begining is\n", x)

x.append(12)
print("The list after appending 12 is\n", x)

print("8 counts {} times".format(x.count(8)))

x.insert(3,15)
print("The list after inserting 15 at position 3 is\n", x)

x.pop()
print("The list after popping last element\n", x)

x.remove(6)
print("The list after removing the element 6 is\n",x)

x.reverse()
print("The reversal of list is\n",x)

x.sort()
print("The sortedlist is \n",x)

print("The index of 4 is",x.index(4))
