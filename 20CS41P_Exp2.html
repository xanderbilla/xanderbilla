Exp1: Program to find factorial of a given number using recursion

            
#***************************************************************#
# Exp2: Program to find factorial of a given number using       #
#       Recursion                                               #
#***************************************************************#

 
def fact(n):
    global f
    if n==1:
        return
    f = f * n
    fact(n - 1)
 
n = int(input("Enter a number: "))
f = 1
fact(n)
print("Factorial of {} is {}".format(n,f))
----------------------------------Output: ----------------------------- 
        
Enter a number: 4
Factorial of 4 is 24
